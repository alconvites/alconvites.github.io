    <script>

        // JavaScript para extrair o nome da URL e adicionar à lista

        document.addEventListener("DOMContentLoaded", function () {

            const listaConvidados = document.getElementById("listaConvidados");


            // Obtém o nome da URL

            const urlParams = new URLSearchParams(window.location.search);

            const nomeConvidado = urlParams.get('nome');


            // Verifica se o nome foi fornecido

            if (nomeConvidado) {

                // Adiciona o nome à lista

                const listItem = document.createElement("li");

                listItem.textContent = nomeConvidado;

                listaConvidados.appendChild(listItem);

            }

        });

    </script>
